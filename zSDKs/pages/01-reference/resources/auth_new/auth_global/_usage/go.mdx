package main

import(
	"context"
	"log"
	"openapi"
	"openapi/pkg/models/shared"
)

func main() {
    s := openapi.New(
        openapi.WithSecurity(shared.Security{
            APIKeyAuth: openapi.String("Token YOUR_API_KEY"),
        }),
        openapi.WithGlobalPathParam(100),
        openapi.WithGlobalQueryParam("some example global query param"),
    )

    ctx := context.Background()
    res, err := s.AuthNew.AuthGlobal(ctx, shared.AuthServiceRequestBody{
        BasicAuth: &shared.AuthServiceRequestBodyBasicAuth{
            Password: "5W6EzvaKCJvc_14",
            Username: "Florian.Champlin60",
        },
        HeaderAuth: []shared.AuthServiceRequestBodyHeaderAuth{
            shared.AuthServiceRequestBodyHeaderAuth{
                ExpectedValue: "interfaces Kalamazoo",
                HeaderName: "vanish Southeast",
            },
        },
    })
    if err != nil {
        log.Fatal(err)
    }

    if res.StatusCode == http.StatusOK {
        // handle response
    }
}
