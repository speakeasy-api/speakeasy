```go showLineNumbers copy filename="Ignores Post usage example"
package main

import(
	"context"
	"log"
	"openapi"
	"openapi/pkg/models/shared"
	"openapi/pkg/models/operations"
)

func main() {
    s := openapi.New(
        openapi.WithSecurity(shared.Security{
            APIKeyAuth: openapi.String("Token YOUR_API_KEY"),
        }),
        openapi.WithGlobalPathParam(100),
        openapi.WithGlobalQueryParam("some example global query param"),
    )

    ctx := context.Background()
    res, err := s.Generation.IgnoresPost(ctx, operations.IgnoresPostRequest{
        RequestBody: operations.IgnoresPostApplicationJSON{
            CallbackURL: openapi.String("https://tall-self-control.org"),
            TestProp: openapi.String("male"),
        },
        TestParam: openapi.String("Borders"),
    })
    if err != nil {
        log.Fatal(err)
    }

    if res.HTTPBinSimpleJSONObject != nil {
        // handle response
    }
}
```
