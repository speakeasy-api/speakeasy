```go showLineNumbers copy filename="Typed Parameter Generation Get usage example"
package main

import(
	"context"
	"log"
	"openapi"
	"openapi/pkg/models/shared"
	"openapi/pkg/models/operations"
	"math/big"
	"openapi/pkg/types"
)

func main() {
    s := openapi.New(
        openapi.WithSecurity(shared.Security{
            APIKeyAuth: openapi.String("Token YOUR_API_KEY"),
        }),
        openapi.WithGlobalPathParam(100),
        openapi.WithGlobalQueryParam("some example global query param"),
    )

    ctx := context.Background()
    res, err := s.Generation.TypedParameterGenerationGet(ctx, operations.TypedParameterGenerationGetRequest{
        Bigint: big.NewInt(485628),
        Date: types.MustDateFromString("2022-09-28"),
        Decimal: types.MustNewDecimalFromString("9774.96"),
        Obj: &operations.TypedParameterGenerationGetObj{
            Bool: false,
            Num: 7875.42,
            Str: "Female following Refined",
        },
    })
    if err != nil {
        log.Fatal(err)
    }

    if res.StatusCode == http.StatusOK {
        // handle response
    }
}
```
