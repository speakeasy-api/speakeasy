```typescript showLineNumbers copy filename="Basic Auth usage example"
import { SDK } from "openapi";
import { BasicAuthResponse, BasicAuthSecurity } from "openapi/dist/sdk/models/operations";

const sdk = new SDK({
  globalPathParam: 100,
  globalQueryParam: "some example global query param",
});
const operationSecurity: BasicAuthSecurity = {
  password: "YOUR_PASSWORD",
  username: "YOUR_USERNAME",
};

sdk.auth.basicAuth({
  passwd: "dqZrTqIEJuENlnS",
  user: "Jaycee_Mante",
}, operationSecurity).then((res: BasicAuthResponse) => {
  if (res.statusCode == 200) {
    // handle response
  }
});
```
