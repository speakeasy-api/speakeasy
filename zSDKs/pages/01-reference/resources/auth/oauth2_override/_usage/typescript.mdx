```typescript showLineNumbers copy filename="Oauth2 Override usage example"
import { SDK } from "openapi";
import { Oauth2OverrideResponse, Oauth2OverrideSecurity } from "openapi/dist/sdk/models/operations";

const sdk = new SDK({
  globalPathParam: 100,
  globalQueryParam: "some example global query param",
});
const operationSecurity: Oauth2OverrideSecurity = {
  oauth2: "Bearer YOUR_OAUTH2_TOKEN",
};

sdk.auth.oauth2Override({}, operationSecurity).then((res: Oauth2OverrideResponse) => {
  if (res.statusCode == 200) {
    // handle response
  }
});
```
