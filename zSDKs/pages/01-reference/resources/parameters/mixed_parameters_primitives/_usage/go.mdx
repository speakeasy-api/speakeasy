```go showLineNumbers copy filename="Mixed Parameters Primitives usage example"
package main

import(
	"context"
	"log"
	"openapi"
	"openapi/pkg/models/shared"
	"openapi/pkg/models/operations"
)

func main() {
    s := openapi.New(
        openapi.WithSecurity(shared.Security{
            APIKeyAuth: openapi.String("Token YOUR_API_KEY"),
        }),
        openapi.WithGlobalPathParam(100),
        openapi.WithGlobalQueryParam("some example global query param"),
    )

    ctx := context.Background()
    res, err := s.Parameters.MixedParametersPrimitives(ctx, operations.MixedParametersPrimitivesRequest{
        HeaderParam: "Shoes Baby male",
        PathParam: "monetize abscond",
        QueryStringParam: "Funk National",
    })
    if err != nil {
        log.Fatal(err)
    }

    if res.Res != nil {
        // handle response
    }
}
```
