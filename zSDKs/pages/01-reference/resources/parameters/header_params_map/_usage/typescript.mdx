```typescript showLineNumbers copy filename="Header Params Map usage example"
import { SDK } from "openapi";
import { HeaderParamsMapResponse } from "openapi/dist/sdk/models/operations";

const sdk = new SDK({
  security: {
    apiKeyAuth: "Token YOUR_API_KEY",
  },
  globalPathParam: 100,
  globalQueryParam: "some example global query param",
});

sdk.parameters.headerParamsMap({
  xHeaderMap: {
    "consequuntur": "Blues",
  },
  xHeaderMapExplode: {
    "perspiciatis": "deposit",
  },
}).then((res: HeaderParamsMapResponse) => {
  if (res.statusCode == 200) {
    // handle response
  }
});
```
