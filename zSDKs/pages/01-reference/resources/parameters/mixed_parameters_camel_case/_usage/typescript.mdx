```typescript showLineNumbers copy filename="Mixed Parameters Camel Case usage example"
import { SDK } from "openapi";
import { MixedParametersCamelCaseResponse } from "openapi/dist/sdk/models/operations";

const sdk = new SDK({
  security: {
    apiKeyAuth: "Token YOUR_API_KEY",
  },
  globalPathParam: 100,
  globalQueryParam: "some example global query param",
});

sdk.parameters.mixedParametersCamelCase({
  headerParam: "Music station",
  pathParam: "eek",
  queryStringParam: "keenly grey Keyboard",
}).then((res: MixedParametersCamelCaseResponse) => {
  if (res.statusCode == 200) {
    // handle response
  }
});
```
