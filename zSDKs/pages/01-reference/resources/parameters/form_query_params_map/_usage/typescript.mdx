```typescript showLineNumbers copy filename="Form Query Params Map usage example"
import { SDK } from "openapi";
import { FormQueryParamsMapResponse } from "openapi/dist/sdk/models/operations";

const sdk = new SDK({
  security: {
    apiKeyAuth: "Token YOUR_API_KEY",
  },
  globalPathParam: 100,
  globalQueryParam: "some example global query param",
});

sdk.parameters.formQueryParamsMap({
  mapParam: {
    "asperiores": "navigating",
  },
  mapParamExploded: {
    "quidem": 806670,
  },
}).then((res: FormQueryParamsMapResponse) => {
  if (res.statusCode == 200) {
    // handle response
  }
});
```
