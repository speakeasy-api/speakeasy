```typescript showLineNumbers copy filename="Deep Object Query Params Map usage example"
import { SDK } from "openapi";
import { DeepObjectQueryParamsMapResponse } from "openapi/dist/sdk/models/operations";

const sdk = new SDK({
  security: {
    apiKeyAuth: "Token YOUR_API_KEY",
  },
  globalPathParam: 100,
  globalQueryParam: "some example global query param",
});

sdk.parameters.deepObjectQueryParamsMap({
  mapArrParam: {
    "laboriosam": [
      "North",
    ],
  },
  mapParam: {
    "deserunt": "payment",
  },
}).then((res: DeepObjectQueryParamsMapResponse) => {
  if (res.statusCode == 200) {
    // handle response
  }
});
```
