```typescript showLineNumbers copy filename="Header Params Primitive usage example"
import { SDK } from "openapi";
import { HeaderParamsPrimitiveResponse } from "openapi/dist/sdk/models/operations";

const sdk = new SDK({
  security: {
    apiKeyAuth: "Token YOUR_API_KEY",
  },
  globalPathParam: 100,
  globalQueryParam: "some example global query param",
});

sdk.parameters.headerParamsPrimitive({
  xHeaderBoolean: false,
  xHeaderInteger: 434761,
  xHeaderNumber: 8980.63,
  xHeaderString: "Legacy",
}).then((res: HeaderParamsPrimitiveResponse) => {
  if (res.statusCode == 200) {
    // handle response
  }
});
```
