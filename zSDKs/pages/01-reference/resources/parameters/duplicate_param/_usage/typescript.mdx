```typescript showLineNumbers copy filename="Duplicate Param usage example"
import { SDK } from "openapi";
import { DuplicateParamResponse } from "openapi/dist/sdk/models/operations";

const sdk = new SDK({
  security: {
    apiKeyAuth: "Token YOUR_API_KEY",
  },
  globalPathParam: 100,
  globalQueryParam: "some example global query param",
});

sdk.parameters.duplicateParam({
  duplicateParamRequest: "azure",
}).then((res: DuplicateParamResponse) => {
  if (res.statusCode == 200) {
    // handle response
  }
});
```
